Hitbox that moves along a Bézier path with an optional cone filter. Cubic and quadratic share one update function. The cone check is just a dot product — no sqrt.

ToleHitbox.lua 
-- cubic / quadratic Bézier, both arities in one function
local function cubicBezier(p0, p1, p2, p3, t)
    local u = 1 - t; local tt = t * t; local uu = u * u
    return (uu * u)     * p0
         + (3 * uu * t) * p1
         + (3 * u * tt) * p2
         + (tt * t)     * p3
end

-- task.defer so the final hit resolves before the hitbox cleans up
function Hitbox:_updateBezier(dt)
    self.BezierElapsed = math.min(self.BezierElapsed + dt, self.BezierDuration)
    local t    = self.BezierElapsed / self.BezierDuration
    local path = self.BezierPath
    local pos  = if #path == 4
        then cubicBezier(path[1], path[2], path[3], path[4], t)
        else quadraticBezier(path[1], path[2], path[3], t)
    if t >= 1 then task.defer(self.destroy, self) end
    return CFrame.new(pos)
end

-- half-angle in degrees, one dot product per target
function Hitbox:_passesCone(pos)
    if not self.ConeAngle or not self.ConeOrigin then return true end
    local cf        = self.ConeOrigin.CFrame
    local toTarget  = (pos - cf.Position).Unit
    local threshold = math.cos(math.rad(self.ConeAngle))
    return cf.LookVector:Dot(toTarget) >= threshold
end

Client-side packet routing via a dispatch table instead of an if/elseif chain. Adding a new packet type is one line. Unknown modes warn so nothing breaks silently.

ToleHitboxLocal.lua 
type Packet = { _Mode: string, _Tick: number, [string]: any }

local active : { [number]: ToleHitbox.Hitbox } = {}

local dispatch : { [string]: (Packet) -> () } = {
    New = function(p)
        local hitbox    = ToleHitbox.new(p :: any)
        active[p._Tick] = hitbox
        ToleHitboxRemote:FireServer(p._Tick)  -- ack to server
    end,
    Start     = function(p) active[p._Tick]:start()              end,
    Stop      = function(p) active[p._Tick]:stop()               end,
    Pause     = function(p) active[p._Tick]:pause()              end,
    Resume    = function(p) active[p._Tick]:resume()             end,
    ClrTag    = function(p) active[p._Tick]:clearTaggedChars()   end,
    AddIgnore = function(p) active[p._Tick]:addIgnore(p.Targets) end,
    Destroy   = function(p)
        if active[p._Tick] then
            active[p._Tick]:destroy()
            active[p._Tick] = nil
        end
    end,
}

ToleHitboxRemote.OnClientEvent:Connect(function(packet : Packet)
    local handler = dispatch[packet._Mode]
    if handler then
        handler(packet)
    else
        warn(("[ToleHitboxLocal] unknown mode: %q"):format(packet._Mode))
    end
end)

Spatial hash grid for fast AOE queries. Skips the remove/insert when an entity stays in the same cell. Squared distance check so no sqrt per candidate.

SpatialHashGrid.lua 
--!strict
local SpatialHashGrid = {}
SpatialHashGrid.__index = SpatialHashGrid

export type Entity = any
type SelfFields = {
    cellSize : number, invSize : number,
    cells       : { [number]: { Entity } },
    entityCells : { [Entity]: number },
    entityPos   : { [Entity]: Vector3 },
}
export type Grid = typeof(setmetatable({} :: SelfFields, SpatialHashGrid))

-- primes for spatial hashing (Teschner et al. 2003)
local function hash(cx : number, cz : number) : number
    return cx * 73856093 + cz * 19349663
end

-- skip remove/insert if entity stayed in the same cell
function SpatialHashGrid.update(self : Grid, entity : Entity, pos : Vector3)
    local oldPos = self.entityPos[entity]
    if not oldPos then return self:insert(entity, pos) end
    local inv = self.invSize
    local cx, cz = math.floor(pos.X * inv),    math.floor(pos.Z * inv)
    local ox, oz = math.floor(oldPos.X * inv), math.floor(oldPos.Z * inv)
    if cx == ox and cz == oz then
        self.entityPos[entity] = pos
        return
    end
    self:remove(entity); self:insert(entity, pos)
end

-- squared distance, no sqrt per candidate
function SpatialHashGrid.queryRadius(
    self : Grid, center : Vector3, radius : number
) : { Entity }
    local inv    = self.invSize
    local r      = math.ceil(radius * inv)
    local cx, cz = math.floor(center.X * inv), math.floor(center.Z * inv)
    local r2     = radius * radius
    local result : { Entity } = {}
    for dx = -r, r do
        for dz = -r, r do
            local cell = self.cells[hash(cx + dx, cz + dz)]
            if cell then
                for _, entity in cell do
                    local d = self.entityPos[entity] - center
                    if d.X * d.X + d.Z * d.Z <= r2 then
                        table.insert(result, entity)
                    end
                end
            end
        end
    end
    return result
end

return SpatialHashGrid

Records character positions at 20Hz so the server can rewind to what the attacker saw at fire-time. Capped at 300ms — clients can't request older snapshots.

SnapshotRewinder.lua 
--!strict
local SNAPSHOT_HZ : number = 20
local MAX_REWIND  : number = 0.300

export type Snapshot = { t: number, cf: CFrame, vel: Vector3 }
type History = { head: number, capacity: number, snapshots: { Snapshot } }

local SnapshotRewinder = {}
local histories : { [Model]: History } = {}

-- interpolates between bracketing snapshots, nil if too old
function SnapshotRewinder.cframeAt(char : Model, t : number) : CFrame?
    local hist = histories[char]
    if not hist then return nil end
    if os.clock() - t > MAX_REWIND then return nil end

    local snaps = hist.snapshots
    local before, after : Snapshot?, Snapshot? = nil, nil
    for i = 1, #snaps do
        local s = snaps[i]
        if s.t <= t and (not before or s.t > before.t) then before = s end
        if s.t >= t and (not after  or s.t < after.t)  then after  = s end
    end

    if before and after then
        local span = after.t - before.t
        if span <= 0 then return before.cf end
        return before.cf:Lerp(after.cf, math.clamp((t - before.t) / span, 0, 1))
    end
    return (before and before.cf) or (after and after.cf)
end

-- true if seenPos is within tolerance of where the target actually was
function SnapshotRewinder.verify(
    target : Model, t : number, seenPos : Vector3, tolerance : number
) : (boolean, CFrame?)
    local cf = SnapshotRewinder.cframeAt(target, t)
    if not cf then return false, nil end
    return (cf.Position - seenPos).Magnitude <= tolerance, cf
end

return table.freeze(SnapshotRewinder)

Effect queue gated by a predicate — built for time-stop combat where hits register mid-stop but apply when time resumes. flush() snapshots before draining so re-entrant pushes go to a fresh queue.

DeferredEffectQueue.lua 
--!strict
local DeferredEffectQueue = {}
DeferredEffectQueue.__index = DeferredEffectQueue

export type Effect<T>  = (T) -> ()
export type Predicate  = () -> boolean

type SelfFields<T> = {
    name : string, queue : { T },
    apply : Effect<T>, isFrozen : Predicate,
}
export type Queue<T> = typeof(setmetatable({} :: SelfFields<T>, DeferredEffectQueue))

function DeferredEffectQueue.new<T>(
    name : string, apply : Effect<T>, isFrozen : Predicate
) : Queue<T>
    return setmetatable({
        name = name, queue = {}, apply = apply, isFrozen = isFrozen,
    }, DeferredEffectQueue) :: any
end

function DeferredEffectQueue.push<T>(self : Queue<T>, effect : T)
    if self.isFrozen() then
        table.insert(self.queue, effect)
    else
        self.apply(effect)
    end
end

-- snapshot before draining so re-entrant pushes go to a fresh queue
function DeferredEffectQueue.flush<T>(self : Queue<T>) : number
    if #self.queue == 0 then return 0 end
    local snapshot = self.queue
    self.queue     = {}
    for _, effect in snapshot do
        local ok, err = pcall(self.apply, effect)
        if not ok then
            warn(string.format("[%s] effect error: %s", self.name, tostring(err)))
        end
    end
    return #snapshot
end

return DeferredEffectQueue

Server-side animation proxy — the server never loads AnimationTracks. Stand modules call :Play() / :Stop() like normal and the proxy fires the RemoteEvent. Lazy __index means no setup per animation key.

StandManager.lua 
local AnimProxy = {}; AnimProxy.__index = AnimProxy

-- auto-creates proxy on first read, no setup needed per key
local AnimTable = {}
AnimTable.__index = function(t, key : string)
    local proxy = setmetatable(
        { _player = t._player, _key = key, IsPlaying = false },
        AnimProxy
    )
    rawset(t, key, proxy)
    return proxy
end

function AnimProxy:Play(fadeTime : number?, weight : number?, speed : number?)
    self.IsPlaying = true
    StandAction:FireClient(self._player, {
        Action   = "PlayAnimation", AnimKey = self._key,
        FadeTime = fadeTime, Weight = weight, Speed = speed,
    })
end

function AnimProxy:Stop(fadeTime : number?)
    self.IsPlaying = false
    StandAction:FireClient(self._player, {
        Action   = "StopAnimation", AnimKey = self._key,
        FadeTime = fadeTime,
    })
end

local function makeAnimTable(player : Player)
    return setmetatable({ _player = player }, AnimTable)
end

DataStore wrapper with MemoryStore session locking. Handles crash-rejoin safely, retries with exponential backoff, and auto-saves on an interval. game:BindToClose flushes everything on shutdown.

SessionManager.lua 
--!strict
local DataStoreService   = game:GetService("DataStoreService")
local MemoryStoreService = game:GetService("MemoryStoreService")
local RunService         = game:GetService("RunService")
local Players            = game:GetService("Players")

local SAVE_INTERVAL = 60
local LOCK_TTL      = 35    -- slightly > SAVE_INTERVAL so lock never expires mid-save
local RETRY_MAX     = 5

local dataStore = DataStoreService:GetDataStore("PlayerData_v1")
local lockMap   = MemoryStoreService:GetSortedMap("SessionLocks")

type Profile = { data: { [string]: any }, dirty: boolean }
local profiles : { [Player]: Profile } = {}

-- exponential backoff, callers don't touch DataStore errors
local function retry(fn: () -> ()): boolean
    for i = 1, RETRY_MAX do
        local ok, err = pcall(fn)
        if ok then return true end
        if i < RETRY_MAX then task.wait(1.5 ^ i) end
        warn(("[Session] attempt %d failed: %s"):format(i, tostring(err)))
    end
    return false
end

-- lock value = JobId so other servers know who holds it
local function acquireLock(userId: number): boolean
    local key = tostring(userId)
    local ok  = pcall(lockMap.SetAsync, lockMap, key, RunService.JobId, LOCK_TTL)
    if not ok then return false end
    return lockMap:GetAsync(key) == RunService.JobId
end

local SessionManager = {}

function SessionManager.load(player: Player)
    if not acquireLock(player.UserId) then
        player:Kick("Already logged in on another server. Please wait 30 seconds.")
        return
    end
    local ok, result = pcall(dataStore.GetAsync, dataStore, tostring(player.UserId))
    if not ok then
        pcall(lockMap.RemoveAsync, lockMap, tostring(player.UserId))
        player:Kick("Data failed to load. Please rejoin.")
        return
    end
    profiles[player] = { data = result or {}, dirty = false }
end

function SessionManager.save(player: Player)
    local p = profiles[player]
    if not p then return end
    retry(function()
        dataStore:SetAsync(tostring(player.UserId), p.data)
    end)
    p.dirty = false
end

function SessionManager.get(player: Player): { [string]: any }?
    local p = profiles[player]
    return p and p.data or nil
end

function SessionManager.unload(player: Player)
    SessionManager.save(player)
    pcall(lockMap.RemoveAsync, lockMap, tostring(player.UserId))
    profiles[player] = nil
end

-- auto-save; lock TTL keeps renewing as a side effect
task.spawn(function()
    while true do
        task.wait(SAVE_INTERVAL)
        for player in profiles do
            task.spawn(SessionManager.save, player)
        end
    end
end)

Players.PlayerRemoving:Connect(SessionManager.unload)
game:BindToClose(function()
    for player in profiles do SessionManager.unload(player) end
end)

return table.freeze(SessionManager)

Token-bucket rate limiter per player. Refills lazily on each check — no heartbeat loop. Pass a higher cost for expensive operations.

RateLimiter.lua 
--!strict
local RateLimiter = {}
RateLimiter.__index = RateLimiter

type Bucket = { tokens: number, lastRefill: number }
type Self   = { rate: number, burst: number, buckets: { [Player]: Bucket } }
export type Limiter = typeof(setmetatable({} :: Self, RateLimiter))

function RateLimiter.new(rate: number, burst: number): Limiter
    return setmetatable({ rate = rate, burst = burst, buckets = {} }, RateLimiter) :: any
end

-- lazy refill, no per-tick loop
local function refill(bucket: Bucket, rate: number, burst: number)
    local now          = os.clock()
    local elapsed      = now - bucket.lastRefill
    bucket.tokens      = math.min(burst, bucket.tokens + elapsed * rate)
    bucket.lastRefill  = now
end

-- false if rate-limited, cost defaults to 1
function RateLimiter.check(self: Limiter, player: Player, cost: number?): boolean
    local c = cost or 1
    local bucket = self.buckets[player]
    if not bucket then
        bucket = { tokens = self.burst, lastRefill = os.clock() }
        self.buckets[player] = bucket
    end
    refill(bucket, self.rate, self.burst)
    if bucket.tokens < c then return false end
    bucket.tokens -= c
    return true
end

function RateLimiter.remove(self: Limiter, player: Player)
    self.buckets[player] = nil
end

return RateLimiter

Sorts services by their declared dependencies and boots in order — all :init() first, then all :start(). One broken service warns instead of crashing everything.

ServiceLoader.lua 
--!strict
type Service = {
    Name         : string,
    Dependencies : { string }?,
    init         : ((any) -> ())?,
    start        : ((any) -> ())?,
}

local registry : { [string]: Service } = {}
local booted   = false

local ServiceLoader = {}

function ServiceLoader.register(service: Service)
    assert(not booted,   "Cannot register after ServiceLoader.boot()")
    assert(service.Name, "Service.Name is required")
    registry[service.Name] = service
end

-- DFS toposort, visited set handles shared deps
local function toposort(): { Service }
    local visited : { [string]: boolean } = {}
    local result  : { Service } = {}
    local function visit(name: string)
        if visited[name] then return end
        visited[name] = true
        local svc = assert(registry[name], "Unknown service: " .. name)
        for _, dep in svc.Dependencies or {} do visit(dep) end
        table.insert(result, svc)
    end
    for name in registry do visit(name) end
    return result
end

function ServiceLoader.boot()
    assert(not booted, "ServiceLoader.boot() called twice")
    booted = true
    local order = toposort()
    -- all init() before any start()
    for _, phase in { "init", "start" } do
        for _, svc in order do
            local fn = (svc :: any)[phase]
            if fn then
                local ok, err = pcall(fn, svc)
                if not ok then
                    warn(("[ServiceLoader] %s:%s() failed: %s"):format(
                        svc.Name, phase, tostring(err)
                    ))
                end
            end
        end
    end
end

return ServiceLoader
