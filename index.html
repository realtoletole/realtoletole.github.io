Spatial hash grid. O(1) average-case AOE and nearest-neighbour queries on hundreds of NPCs. Cells keyed by an integer hash so lookups stay fast. Update path skips work when the entity hasn't crossed a cell boundary, queries test squared distance to avoid a sqrt per candidate. 

--!strict
local SpatialHashGrid = {}
SpatialHashGrid.__index = SpatialHashGrid

export type Entity = any
type SelfFields = {
    cellSize : number, invSize : number,
    cells       : { [number]: { Entity } },
    entityCells : { [Entity]: number },
    entityPos   : { [Entity]: Vector3 },
}
export type Grid = typeof(setmetatable({} :: SelfFields, SpatialHashGrid))

-- Famous primes used for spatial hashing (Teschner et al. 2003).
local function hash(cx : number, cz : number) : number
    return cx * 73856093 + cz * 19349663
end

function SpatialHashGrid.new(cellSize : number) : Grid
    return setmetatable({
        cellSize = cellSize, invSize = 1 / cellSize,
        cells = {}, entityCells = {}, entityPos = {},
    }, SpatialHashGrid) :: any
end

-- Cheap path: if the entity is still in the same cell, just patch position.
function SpatialHashGrid.update(self : Grid, entity : Entity, pos : Vector3)
    local oldPos = self.entityPos[entity]
    if not oldPos then return self:insert(entity, pos) end
    local inv = self.invSize
    local cx, cz = math.floor(pos.X * inv),    math.floor(pos.Z * inv)
    local ox, oz = math.floor(oldPos.X * inv), math.floor(oldPos.Z * inv)
    if cx == ox and cz == oz then
        self.entityPos[entity] = pos
        return
    end
    self:remove(entity); self:insert(entity, pos)
end

-- Iterate only the cells the query disc could touch, then narrow by
-- squared distance to skip a sqrt per candidate.
function SpatialHashGrid.queryRadius(
    self : Grid, center : Vector3, radius : number
) : { Entity }
    local inv = self.invSize
    local r   = math.ceil(radius * inv)
    local cx, cz = math.floor(center.X * inv), math.floor(center.Z * inv)
    local r2 = radius * radius
    local result : { Entity } = {}
    for dx = -r, r do
        for dz = -r, r do
            local cell = self.cells[hash(cx + dx, cz + dz)]
            if cell then
                for _, entity in cell do
                    local d = self.entityPos[entity] - center
                    if d.X * d.X + d.Z * d.Z <= r2 then
                        table.insert(result, entity)
                    end
                end
            end
        end
    end
    return result
end

return SpatialHashGrid

Lag compensation. Records each character's position 20 times a second so the server can check what the attacker actually saw at fire-time. Bounded at 300 ms. Strict-typed, ring-buffered for constant memory. 

--!strict
local SNAPSHOT_HZ : number = 20
local MAX_REWIND  : number = 0.300

export type Snapshot = { t: number, cf: CFrame, vel: Vector3 }
type History = { head: number, capacity: number, snapshots: { Snapshot } }

local SnapshotRewinder = {}
local histories : { [Model]: History } = {}

-- Returns char's CFrame at clock time `t`, interpolated between the two
-- bracketing snapshots. nil if older than MAX_REWIND.
function SnapshotRewinder.cframeAt(char : Model, t : number) : CFrame?
    local hist = histories[char]
    if not hist then return nil end
    if os.clock() - t > MAX_REWIND then return nil end

    local snaps = hist.snapshots
    local before, after : Snapshot?, Snapshot? = nil, nil
    for i = 1, #snaps do
        local s = snaps[i]
        if s.t <= t and (not before or s.t > before.t) then before = s end
        if s.t >= t and (not after  or s.t < after.t)  then after  = s end
    end

    if before and after then
        local span = after.t - before.t
        if span <= 0 then return before.cf end
        return before.cf:Lerp(after.cf, math.clamp((t - before.t) / span, 0, 1))
    end
    return (before and before.cf) or (after and after.cf)
end

-- Validates an attacker's reported aim. (true, rewoundCFrame) if `seenPos`
-- is within `tolerance` studs of where target really was at clock time t.
function SnapshotRewinder.verify(
    target : Model, t : number, seenPos : Vector3, tolerance : number
) : (boolean, CFrame?)
    local cf = SnapshotRewinder.cframeAt(target, t)
    if not cf then return false, nil end
    return (cf.Position - seenPos).Magnitude <= tolerance, cf
end

return table.freeze(SnapshotRewinder)

Cancels in-flight attacks. Each attack gets a unique ID. Calling Feint() bumps that ID, and any pending callback holding the old ID becomes a no-op. No race conditions, no orphan hitboxes. 

function FeintSystem.RegisterAttackStart(char, attackType, SkillName, attackData)
    -- Bump per-character counter; this id is what every async callback
    -- must check before it applies any effect.
    FeintSystem.AttackIds[char] = (FeintSystem.AttackIds[char] or 0) + 1
    local attackId = FeintSystem.AttackIds[char]

    FeintSystem.AttackStartTimes[char] = {
        StartTime  = os.clock(),
        AttackType = attackType,
        SkillName  = SkillName,
        AttackId   = attackId,
    }

    attackData = attackData or {}
    attackData.AttackId = attackId
    FeintSystem.ActiveAttacks[char] = attackData
end

function FeintSystem.Feint(player)
    local char = player.Character
    if not char or not FeintSystem.CanFeint(char) then return false end

    -- Invalidate every in-flight callback for this attack in one line.
    FeintSystem.AttackIds[char] = (FeintSystem.AttackIds[char] or 0) + 1

    local data = FeintSystem.ActiveAttacks[char]
    if data then
        for _, anim in ipairs(data.Animations or {}) do
            if anim.IsPlaying then anim:Stop() end
        end
        for _, hb in ipairs(data.Hitboxes or {}) do
            if hb.Stop then hb:Stop() end
        end
        for _, vfx in ipairs(data.VFX or {}) do
            if vfx and vfx.Parent then vfx:Destroy() end
        end
        if data.CleanupFunction then pcall(data.CleanupFunction) end
    end

    -- Strip body movers, reset walk/jump, arm cooldown, clear state.
    local root = char:FindFirstChild("HumanoidRootPart")
    if root then
        for _, v in ipairs(root:GetChildren()) do
            if v:IsA("BodyVelocity") or v:IsA("LinearVelocity") then v:Destroy() end
        end
    end
    setFeintCooldown(char, player)
    FeintSystem.ActiveAttacks[char] = nil
    return true
end
