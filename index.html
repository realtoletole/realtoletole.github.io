Server-side lag compensation — records HumanoidRootPart snapshots at a fixed cadence and lets the server "rewind" a target's transform back to what the attacker actually saw at fire-time. Bounded by MAX_REWIND so a malicious client cannot request rewinds arbitrarily far back. Strict-typed, ring-buffered, frozen module — single Heartbeat drives all snapshots. 

--!strict
local SNAPSHOT_HZ : number = 20
local MAX_REWIND  : number = 0.300

export type Snapshot = { t: number, cf: CFrame, vel: Vector3 }
type History = { head: number, capacity: number, snapshots: { Snapshot } }

local SnapshotRewinder = {}
local histories : { [Model]: History } = {}

-- Returns char's CFrame at clock time `t`, linearly interpolated between
-- the two bracketing snapshots. nil if older than MAX_REWIND.
function SnapshotRewinder.cframeAt(char : Model, t : number) : CFrame?
    local hist = histories[char]
    if not hist then return nil end
    if os.clock() - t > MAX_REWIND then return nil end

    local snaps = hist.snapshots
    local before, after : Snapshot?, Snapshot? = nil, nil
    for i = 1, #snaps do
        local s = snaps[i]
        if s.t <= t and (not before or s.t > before.t) then before = s end
        if s.t >= t and (not after  or s.t < after.t)  then after  = s end
    end

    if before and after then
        local span = after.t - before.t
        if span <= 0 then return before.cf end
        return before.cf:Lerp(after.cf, math.clamp((t - before.t) / span, 0, 1))
    end
    return (before and before.cf) or (after and after.cf)
end

-- Validates an attacker's reported aim. (true, rewoundCFrame) if `seenPos`
-- is within `tolerance` studs of where target really was at clock time t.
function SnapshotRewinder.verify(
    target : Model, t : number, seenPos : Vector3, tolerance : number
) : (boolean, CFrame?)
    local cf = SnapshotRewinder.cframeAt(target, t)
    if not cf then return false, nil end
    return (cf.Position - seenPos).Magnitude <= tolerance, cf
end

return table.freeze(SnapshotRewinder)

From FeintSystem — attack cancellation via incrementing AttackId. Every started attack stamps an ID into a per-character counter; the in-flight attack carries that ID through animations, hitboxes, and VFX. Calling Feint() bumps the counter, instantly invalidating the attack — any pending callback that doesn't match the current ID becomes a no-op. No race conditions, no lingering hitboxes, no orphan VFX. 

function FeintSystem.RegisterAttackStart(char, attackType, SkillName, attackData)
    -- Bump per-character attack counter; this id is what every async
    -- callback must check before applying its effect.
    FeintSystem.AttackIds[char] = (FeintSystem.AttackIds[char] or 0) + 1
    local attackId = FeintSystem.AttackIds[char]

    FeintSystem.AttackStartTimes[char] = {
        StartTime  = os.clock(),
        AttackType = attackType,
        SkillName  = SkillName,
        AttackId   = attackId,
    }

    attackData = attackData or {}
    attackData.AttackId = attackId
    FeintSystem.ActiveAttacks[char] = attackData
end

function FeintSystem.Feint(player)
    local char = player.Character
    if not char or not FeintSystem.CanFeint(char) then return false end

    -- Invalidate every in-flight callback for this attack in one line.
    FeintSystem.AttackIds[char] = (FeintSystem.AttackIds[char] or 0) + 1

    local data = FeintSystem.ActiveAttacks[char]
    if data then
        for _, anim in ipairs(data.Animations or {}) do
            if anim.IsPlaying then anim:Stop() end
        end
        for _, hb in ipairs(data.Hitboxes or {}) do
            if hb.Stop then hb:Stop() end
        end
        for _, vfx in ipairs(data.VFX or {}) do
            if vfx and vfx.Parent then vfx:Destroy() end
        end
        if data.CleanupFunction then pcall(data.CleanupFunction) end
    end

    -- Strip body movers; reset walk/jump; arm cooldown; clear state.
    local root = char:FindFirstChild("HumanoidRootPart")
    if root then
        for _, v in ipairs(root:GetChildren()) do
            if v:IsA("BodyVelocity") or v:IsA("LinearVelocity") then v:Destroy() end
        end
    end
    setFeintCooldown(char, player)
    FeintSystem.ActiveAttacks[char] = nil
    return true
end

From KnockdownService.Carry — a knocked player can be picked up and carried. Locks the victim with a Weld, swaps their parts to a non-colliding group, marks them Massless so the carrier doesn't drag, plays paired animations on both characters, and listens on Heartbeat for the carrier becoming stunned — at which point the carry releases automatically. 

function KnockdownService.Carry(carrier, victim)
    if not KnockdownService.IsKnocked(victim) then return false end
    local data = KnockdownService.KnockedPlayers[victim]
    if data.carriedBy then return false end

    local carrierTorso = carrier:FindFirstChild("Torso") or carrier.HumanoidRootPart
    local victimRoot   = victim:FindFirstChild("HumanoidRootPart")
    if (carrier.HumanoidRootPart.Position - victimRoot.Position).Magnitude > 7 then
        return false  -- range gate
    end

    -- Cancel the auto-wakeup; lift the ragdoll so the weld can pose them.
    if data.thread then pcall(task.cancel, data.thread); data.thread = nil end
    CM.Ragdoll.Remove(victim)

    -- Persist original Massless so we can restore on release.
    for _, part in ipairs(victim:GetChildren()) do
        if part:IsA("BasePart") then
            part:SetAttribute("WasMassless", part.Massless)
            part.Massless       = true
            part.CollisionGroup = "Carried"
        end
    end

    local weld  = Instance.new("Weld")
    weld.Part0  = carrierTorso
    weld.Part1  = victimRoot
    weld.C0     = CFrame.new(-0.5, 0.25, 0) * CFrame.Angles(360, math.pi, 0)
    weld.Parent = carrier
    weld.Name   = "CarryWeld"

    data.carriedBy, data.carryWeld = carrier, weld
    carrier:SetAttribute("Carrying", victim.Name)
    victim:SetAttribute("Carried",   carrier.Name)

    -- Heartbeat watcher: if the carrier gets stunned, drop the victim.
    local carrierCV = CM.Table[carrier]
    if carrierCV then
        task.delay(0.5, function()
            if not data.carriedBy then return end
            local conn; conn = RunService.Heartbeat:Connect(function()
                if not carrier.Parent then conn:Disconnect(); return end
                if carrierCV.Stunned then
                    conn:Disconnect()
                    KnockdownService.ReleaseCarry(carrier, victim)
                end
            end)
            data.hitConn = conn
        end)
    end
    return true
end
